Corelation

clc
clear
Data = load('4.txt')*1e5;

sig = std(Data);
mm = mean(Data);
N = length(Data);
sigma_0 = std(Data);

Auto-Corelation

for Tau = 1:N
    d1 = (Data-mm)/sigma_0;
    d2 = d1(Tau:N);
    c(Tau) = sum(d1(1:N-Tau+1).*d2)/N;
end
figure(1);ax = gca;


plot((1:N)*0.02,c,'.')
hold on
plot((1:N)*0.02,zeros(N,1),'k')
title('Auto-Corelation','fontsize',18)
xlabel('\tau','fontsize',18)
ylabel('C_{xx}(\tau)','fontsize',18)
ax.FontSize = 16;
ax.LineWidth = 1.5;
axis([0 40 -0.2 1])
hold off

Cross-Corelation

Dif=diff(Data);

for Tau = 1:N-1
    c_xy(Tau) =  cross_correlation(Data, Dif, Tau);
end
figure(2);ax = gca;

plot((1:length(c_xy))*0.02,c_xy,'.')
hold on
plot((1:N)*0.02,zeros(N,1),'k')
title('Cross-Corelation','fontsize',18)
xlabel('\tau','fontsize',18)
ylabel('C_{x\eta}(\tau)','fontsize',18)
ax.FontSize = 16;
ax.LineWidth = 1.5;
axis([0 40 -0.2 0.05])
hold off
